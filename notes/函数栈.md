## gdb disas 代码例子(x86_64)
```cpp
#include <stdio.h>
int test(int a, int b)
{
	int add = a + b;
	int mul = a * b;
	int div = a / ((b == 0)? 1 : b);
	int sub = a - b;

	int res = add + mul + div + sub;
	return res;
}

int main()
{
	int p1 = 5, p2=6;
	int res = test(p1,p2);
	printf("%d",res);
    return 0;
}
```
